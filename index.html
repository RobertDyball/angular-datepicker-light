Example 1: Show on Focus

HTML
<body ng-app="mainApp">
    <div ng-controller="OnFocusCtrl as vm">
        <input type="text" ng-model="vm.date" />
    </div>
</body>
JavaScript
angular
    .module('mainApp', ['datepickerLightModule'])
    .controller('OnFocusCtrl', function () {
        var vm = this;
        
        vm.date = dateText;
    });

Example 2: Show Inline

HTML
<body ng-app="mainApp">
    <div ng-controller="InlineCtrl as vm">
        <input type="text"
               ng-model="vm.date"   
               angular-datepicker-light="vm.datepickerOptions" />
    </div>
</body>
JavaScript
angular
    .module('mainApp', ['datepickerLightModule'])
    .controller('InlineCtrl', InlineCtrl)
        
    function InlineCtrl(){
        var vm = this;
        
        vm.date = dateText;
        
        vm.datepickerOptions = {
            inline: true
        };
    }

Example 4: Disabled Dates

HTML
<body ng-app="mainApp">
    <div ng-controller="InlineDisabledDatesCtrl as vm">
        <input type="text"
               ng-model="vm.date"   
               angular-datepicker-light="vm.datepickerOptions" />
    </div>
</body>
JavaScript
angular
    .module('mainApp', ['datepickerLightModule'])
    .controller('InlineDisabledDatesCtrl', InlineDisabledDatesCtrl);
    
    function InlineDisabledDatesCtrl() {
        var vm = this;
        
        vm.date = dateText;
        
        vm.datepickerOptions = {
            inline: true,
            renderDate: function (e) {
                var date = e.date.getDate();
                var enable = (date < 5 || date > 15);
                return {
                    enabled: enable,
                    tooltip: enable ? null : "We are closed!"
                }
            }
        };
    }
